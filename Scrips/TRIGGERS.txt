----------------------------------------------------------------------------------------------
--TRIGGERS
----------------------------------------------------------------------------------------------
--TABLE PERSON
----------------------------------------------------------------------------------------------
CREATE OR REPLACE TRIGGER us.beforeInsertPerson
BEFORE INSERT
ON usu.person
FOR EACH ROW
BEGIN
    :new.creation_date:=SYSDATE;
    :new.creation_user:=USER;
END;

CREATE OR REPLACE TRIGGER usu.beforeUpdatePerson
BEFORE UPDATE
ON usu.person
FOR EACH ROW
BEGIN
    :new.last_update_date:=SYSDATE;
    :new.last_update_user:=USER;
END;
